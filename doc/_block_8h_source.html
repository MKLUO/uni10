316  template<typename T, typename U>
317  uni10_bool operator== (const Block<T>& kblk1, const Block<U>& kblk2){
318 
319  if( (kblk1.row_ != kblk2.row_) || (kblk1.col_ != kblk2.col_) || (kblk1.diag_ != kblk2.diag_) )
320  return false;
321 
322  for(uni10_int i = 0; i < kblk1.elem_.elem_num_; i++)
323  if(UNI10_REAL(kblk1.elem_.elem_ptr_[i] )- UNI10_REAL(kblk2.elem_.elem_ptr_[i] )> 10E-12)
324  return false;
325 
326  if(kblk1.elem_.uni10_typeid_ == 2)
327  for(uni10_int i = 0; i < kblk1.elem_.elem_num_; i++)
328  if(UNI10_IMAG(kblk1.elem_.elem_ptr_[i]) - UNI10_IMAG(kblk2.elem_.elem_ptr_[i]) > 10E-12)
329  return false;
330 
331  return true;
332  }
