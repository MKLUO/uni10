663  template<typename uni10_type>
664  std::ostream& operator<< (std::ostream& os, const UniTensor<uni10_type>& UniT){
665 
666  if(UniT.paras == NULL){
667  std::cout<<"This UniTensor has not been initialized."<< std::endl;
668  return os;
669  }
670 
671  if(!(*(UniT.status) & UniT.HAVEBOND)){
672  if(UniT.U_elem->ongpu_)
673  std::cout<<"\nScalar: " << UniT.U_elem->elem_ptr_[0]<<", onGPU";
674  else{
675  std::cout.precision(10);
676  std::cout.setf(std::ios::fixed, std::ios::floatfield);
677  std::cout<<"\nScalar: " << UniT.U_elem->elem_ptr_[0];
678  }
679  std::cout<<"\n\n";
680  return os;
681  }
682 
683  uni10_uint64 row = 0;
684  uni10_uint64 col = 0;
685 
686  std::vector<Bond>bonds = (*UniT.bonds);
687  for(uni10_uint64 i = 0; i < bonds.size(); i++)
688  if(bonds[i].type() == BD_IN)
689  row++;
690  else
691  col++;
692  uni10_uint64 layer = std::max(row, col);
693  uni10_uint64 nmlen = UniT.name->length() + 2;
694  uni10_int star = 12 + (14 - nmlen) / 2;
695  std::cout << std::endl;
696  for(uni10_int s = 0; s < star; s++)
697  std::cout << "*";
698  if(UniT.name->length() > 0)
699  std::cout << " " << (*UniT.name) << " ";
700  for(uni10_int s = 0; s < star; s++)
701  std::cout <<"*";
702  std::cout<<std::endl;
703 
704  if(UniT.U_elem->uni10_typeid_ == 1)
705  std::cout << "REAL" << std::endl;
706  else if(UniT.U_elem->uni10_typeid_ == 2)
707  std::cout << "COMPLEX" << std::endl;
708 
709  if(UniT.U_elem->ongpu_)
710  std::cout<<"\n onGPU";
711  std::cout << "\n ____________\n";
712  std::cout << " | |\n";
713  uni10_uint64 llab = 0;
714  uni10_uint64 rlab = 0;
715  char buf[128];
716  for(uni10_uint64 l = 0; l < layer; l++){
717  if(l < row && l < col){
718  llab = (*UniT.labels)[l];
719  rlab = (*UniT.labels)[row + l];
720  sprintf(buf, " %5ld___|%-4d %4d|___%-5ld\n", llab, bonds[l].dim(), bonds[row + l].dim(), rlab);
721  std::cout<<buf;
722  }
723  else if(l < row){
724  llab = (*UniT.labels)[l];
725  sprintf(buf, " %5ld___|%-4d %4s|\n", llab, bonds[l].dim(), "");
726  std::cout<<buf;
727  }
728  else if(l < col){
729  rlab = (*UniT.labels)[row + l];
730  sprintf(buf, " %5s |%4s %4d|___%-5ld\n", "", "", bonds[row + l].dim(), rlab);
731  std::cout << buf;
732  }
733  std::cout << " | | \n";
734  }
735  std::cout << " |____________|\n";
736 
737  std::cout << "\n================BONDS===============\n";
738  for(uni10_uint64 b = 0; b < bonds.size(); b++)
739  std::cout << bonds[b];
740 
741  std::cout<<"\n===============BLOCKS===============\n";
742  typename std::map<Qnum, Block<uni10_type> >::const_iterator it = UniT.blocks->begin();
743  for (; it != UniT.blocks->end(); it++ ){
744  std::cout << "--- " << it->first << ": ";// << Rnum << " x " << Cnum << " = " << Rnum * Cnum << " ---\n\n";
745  if((*(UniT.status) & UniT.HAVEELEM))
746  std::cout<<it->second;
747  else
748  std::cout<<it->second.row() << " x "<<it->second.col()<<": "<<it->second.ElemNum()<<std::endl<<std::endl;
749  }
750  std::cout << "Total elemNum: "<<(*UniT.U_elemNum)<<std::endl;
751  std::cout << "====================================" << std::endl;
752  os << "\n";
753  return os;
754  }
